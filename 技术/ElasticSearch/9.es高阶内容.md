## Term Vector

term vector一般用于做一些数据做探查，获取document中的某个field内的各个分词的统计信息（term statistics和field statistics并不精准，不会考虑有的doc可能被删除情况）

- field statistics
  1. document count，有多少document包含这个field; 
  2. sum of document frequency，field中所有term的df之和; 
  3. sum of total term frequency，field中的所有term的tf之和
- term statistics
  1. total term frequency, 一个term在所有document中出现的频率; 
  2. document frequency，有多少document包含这个term
- term information : term frequency in the field, term positions, start and end offsets, term payloads

term vector，涉及了很多的term和field相关的统计信息，有两种方式可以采集到这个统计信息

1. index-time，在mapping里配置一下，然后建立索引的时候，就直接生成这些term和field的统计信息

   ```json
   PUT /my_index
   {
       "mappings": {
           "my_type": {
               "properties": {
                   "my_field": {
                       "type": "text",
                       "term_vector": "with_positions_offsets_payloads",
                       "store" : true,
                       "analyzer" : "fulltext_analyzer"
                   },
                   "fullname": {
                       "type": "text",
                       "analyzer" : "fulltext_analyzer"
                   }
                }
            }
        }
   }
   ```



   ```json
   GET /my_index/my_type/1/_termvectors
   {
       "fields" : ["my_field"],
       "offsets" : true,
       "payloads" : true,
       "positions" : true,
       "term_statistics" : true,
       "field_statistics" : true
   }
   
   {
       "_index": "my_index",
       "_type": "my_type",
       "_id": "1",
       "_version": 1,
       "found": true,
       "took": 10,
       "term_vectors": {
           "my_field": {
               "field_statistics": {
                   # 所有doc中的所有term的doc_freq
                   "sum_doc_freq": 6,
                   # 有多少document包含这个field
                   "doc_count": 2,
                   # 所有doc中的所有term的ttf
                   "sum_ttf": 8
               },
               # 这里就是所有分词的term statistics情况
               "terms": {
                   "hello": {
               		# 有多少document包含这个term
                       "doc_freq": 2,
               		# 一个term在所有document中出现的频率
                       "ttf": 2,
               		# 分词在当前field出现的频率 
                       "term_freq": 1,
                       "tokens": [
               			# 多个hello分词分别出现的位置等信息
                           {
                               "position": 0,
                               "start_offset": 0,
                               "end_offset": 5,
                               "payload": "d29yZA=="
                           }
                       ]
                   },
                   "test": {
                       "doc_freq": 2,
                       "ttf": 4,
                       "term_freq": 3,
                       "tokens": [
                           {
                               "position": 1,
                               "start_offset": 6,
                               "end_offset": 10,
                               "payload": "d29yZA=="
                           },
                           ...
                       ]
                   }
               }
           }
       }
   }
   ```

2. query-time，之前没有生成过任何的Term vector信息，然后在查看term vector的时候，现场计算出各种统计信息返回出来

   ```json
   GET /my_index/my_type/1/_termvectors
   {
       "fields" : ["fullname"],
       "offsets" : true,
       "positions" : true,
       "term_statistics" : true,
       "field_statistics" : true
   }
   ```

   一般来说，如果条件允许，用query-time的term vector就可以了，现场去探查一下就好了

### 手动指定doc的term vector

```json
GET /my_index/my_type/_termvectors
{
    # 手动指定的doc的计数不会统计在最后的结果中
    "doc" : {
        "fullname" : "Leo Li",
        # 会对text中的内容进行分词，然后计算这些分词在现有的所有doc中的一些统计信息
        # 这个比较有用的点是可以手动指定要探查的term的数据情况
        "text" : "hello test test test"
    },
    "fields" : ["my_field"],
    "offsets" : true,
    "payloads" : true,
    "positions" : true,
    "term_statistics" : true,
    "field_statistics" : true,
	# 可以手动指定分词器
	"per_field_analyzer" : {
        "my_field": "standard"
    },
	# 可以过滤掉一些频率过低的分词
	"filter" : {
        # 展示最多3个分词
        "max_num_terms" : 3,
        # 分词在当前field出现的频率最小为1
        "min_term_freq" : 1,
        # 最少有1个doc包含这个分词
        "min_doc_freq" : 1
    }
}
```

###批量查询Term Vector

```json
GET /my_index/_mtermvectors
{
   "docs": [
      {
         "_type": "test",
         "_id": "2",
         "fields": [
            "text"
         ],
         "term_statistics": true
      },
      {
         "_type": "test",
         "_id": "1"

      {
         "_index": "my_test",
         "_type": "my_type",
         "doc" : {
           "fullname" : "Leo Li",
           "text" : "other hello test ..."
         }
      }
   ]
}
```



## 地理位置

```json
PUT /my_index 
{
    "mappings": {
        "my_type": {
            "properties": {
                "location": {
                    # geo_point 指定地理位置
                    "type": "geo_point"
                }
            }
        }
    }
}

PUT my_index/my_type/1
{
    "text": "Geo-point as an object",
    "location": { 
        # latitude 纬度
        "lat": 41.12,
        # longitude 经度
        "lon": -71.34
    }
}
```

### 地理位置查询

```json
GET /my_index/my_type/_search 
{
    "query": {
        # geo_bounding_box查询，查询某个矩形的地理位置范围内的坐标点
        "geo_bounding_box": {
            "location": {
        		# 矩形的左上角坐标点
                "top_left": {
                    "lat": 42,
                    "lon": -72
                },
    			# 矩形的右下角坐标点
                "bottom_right": {
                    "lat": 40,
                    "lon": -74
                }
            }
        }
    }
}

GET /hotel_app/hotels/_search 
{
    "query": {
        "bool": {
            "must": [
                {
                    "match_all": {}
                }
            ],
            "filter": {
                # geo_polygon搜索多边形内的点
                "geo_polygon": {
                    "location": {
                        "points": [
                            {"lat" : 40.73, "lon" : -74.1},
                            {"lat" : 40.01, "lon" : -71.12},
                            {"lat" : 50.56, "lon" : -90.58}
                        ]
                    }
                }
            }
        }
    }
}

GET /hotel_app/hotels/_search
{
    "query": {
        "bool": {
            "must": [
                {
                    "match_all": {}
                }
            ],
            "filter": {
                # geo_distance查询距离指定位置，指定距离内的坐标点
                "geo_distance": {
                    "distance": "200km",
                    "location": {
                        "lat": 40,
                        "lon": -70
                    }
                }
            }
        }
    }
}
```

### 地理位置聚合分析

```json
GET /hotel_app/hotels/_search
{
    "size": 0,
    "aggs": {
        "agg_by_distance_range": {
            # aggs-geo_distance距离范围的聚合
            "geo_distance": {
                "field": "location",
            	"unit": "mi", 
            	# 统计距离[40, -70]几个范围内的酒店的数量
                "origin": {
                    "lat": 40,
                    "lon": -70
                },
        		# 0~100m有几个酒店，100m~300m有几个酒店，300m以上有几个酒店
                "ranges": [
                    {
                        "to": 100
                    },
                    {
                        "from": 100,
                        "to": 300
                    },
                    {
                        "from": 300
                    }
                ]
            }
        }
    }
}
```
